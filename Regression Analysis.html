#Creating a function that will return the slope and intercept for our x and y variables.
def slope_intercept(x_val, y_val):
    x = np.array(x_val)
    y = np.array(y_val)
    m = (((np.mean(x)*np.mean(y)) - np.mean(x*y)) / 
       ((np.mean(x)*np.mean(x)) - np.mean(x*x)))
    m = round(m,2)
    b = (np.mean(y) - np.mean(x)*m)
    b = round(b,2)
    return m,b

slope_intercept(x,y)

#A function that returns the root mean square error.
#The RMSE measures how spread out the residuals (prediction errors) are.
def rmse(y1,y_hat):
    y_actual = np.array(y1)
    y_pred = np.array(y_hat)
    error = (y_actual - y_pred)**2
    error_mean = round(np.mean(error))
    error_sq = np.sqrt(error_mean)
    return error_sq
